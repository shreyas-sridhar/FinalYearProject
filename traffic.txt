import RPi.GPIO as GPIO
from mfrc522 import MFRC522  # For SPI-based RFID reader (RFID2)
import serial
import time
import threading
from datetime import datetime

# GPIO setup
GPIO.setmode(GPIO.BOARD)
GPIO.setwarnings(False)

# Define LED pins as per your connections
LEDs = {
    "Signal1": {
        "Red": 5,
        "Yellow": 12,
        "Green": 3,
        "White": 40
    },
    "Signal2": {
        "Red": 13,
        "Yellow": 38,
        "Green": 15,
        "White": 16
    }
}

# Setup LED pins
for signal in LEDs.values():
    for pin in signal.values():
        GPIO.setup(pin, GPIO.OUT)
        GPIO.output(pin, GPIO.LOW)

# Setup RFID2 (MFRC522) via SPI CE1
reader2 = MFRC522(bus=0, device=1)

# Setup RFID1 via Arduino Serial (/dev/ttyUSB0)
try:
    serial_rfid1 = serial.Serial('/dev/ttyUSB0', 9600, timeout=1)
    time.sleep(2)  # wait for Arduino reset
except Exception as e:
    print("Failed to connect to Arduino:", e)
    serial_rfid1 = None

# Shared state
paused = False
interrupted_signal = None
interrupted_time = 0
shutdown = False
lock = threading.Lock()

# RFID1 Listener from Arduino
def arduino_rfid_listener():
    global paused, interrupted_signal, interrupted_time, shutdown
    while not shutdown and serial_rfid1:
        try:
            if serial_rfid1.in_waiting > 0:
                data = serial_rfid1.readline().decode('utf-8').strip()
                if data:
                    with lock:
                        print(f"[{datetime.now()}] Signal1 RFID from Arduino: {data}")
                        paused = True
                        interrupted_signal = "Signal1"
                        interrupted_time = time.time()
                        for color in ["Red", "Yellow", "Green"]:
                            GPIO.output(LEDs["Signal1"][color], GPIO.LOW)
                            GPIO.output(LEDs["Signal2"][color], GPIO.LOW)
                        GPIO.output(LEDs["Signal1"]["White"], GPIO.HIGH)
            time.sleep(0.1)
        except Exception as e:
            print("Arduino RFID Error:", e)

# RFID2 Listener from MFRC522
def mfrc_rfid_listener(reader, signal_name):
    global paused, interrupted_signal, interrupted_time, shutdown
    while not shutdown:
        try:
            (status, _) = reader.MFRC522_Request(reader.PICC_REQIDL)
            if status == reader.MI_OK:
                (status, uid) = reader.MFRC522_Anticoll()
                if status == reader.MI_OK:
                    id = int.from_bytes(bytes(uid), "big")
                    with lock:
                        print(f"[{datetime.now()}] {signal_name} RFID detected: {id}")
                        paused = True
                        interrupted_signal = signal_name
                        interrupted_time = time.time()
                        for color in ["Red", "Yellow", "Green"]:
                            GPIO.output(LEDs["Signal1"][color], GPIO.LOW)
                            GPIO.output(LEDs["Signal2"][color], GPIO.LOW)
                        GPIO.output(LEDs[signal_name]["White"], GPIO.HIGH)
            time.sleep(0.1)
        except Exception as e:
            print(f"{signal_name} RFID Error:", e)

# Traffic Light Logic
def traffic_light_controller():
    global paused, interrupted_signal, interrupted_time, shutdown
    current_signal = "Signal1"
    last_state = {"Signal1": "Red", "Signal2": "Red"}
    while not shutdown:
        if not paused:
            new_state = {"Signal1": "Red", "Signal2": "Red"}
            new_state[current_signal] = "Green"
            for sig in ["Signal1", "Signal2"]:
                if new_state[sig] != last_state[sig]:
                    print(f"[{datetime.now()}] {sig}: {last_state[sig]} → {new_state[sig]}")
                    last_state[sig] = new_state[sig]
            for sig in ["Signal1", "Signal2"]:
                GPIO.output(LEDs[sig]["Red"], new_state[sig] == "Red")
                GPIO.output(LEDs[sig]["Green"], new_state[sig] == "Green")
                GPIO.output(LEDs[sig]["Yellow"], GPIO.LOW)
            time.sleep(5)
            GPIO.output(LEDs[current_signal]["Green"], GPIO.LOW)
            GPIO.output(LEDs[current_signal]["Yellow"], GPIO.HIGH)
            time.sleep(2)
            current_signal = "Signal2" if current_signal == "Signal1" else "Signal1"
        else:
            with lock:
                if time.time() - interrupted_time >= 10:
                    paused = False
                    for sig in ["Signal1", "Signal2"]:
                        GPIO.output(LEDs[sig]["White"], GPIO.LOW)
                        GPIO.output(LEDs[sig]["Red"], GPIO.HIGH)
                    last_state = {"Signal1": "Red", "Signal2": "Red"}
        time.sleep(0.1)

# Stop listener
def stop_listener():
    global shutdown
    while True:
        cmd = input().strip()
        if cmd.lower() == "stop":
            shutdown = True
            break

# Launch all threads
threading.Thread(target=arduino_rfid_listener).start()
threading.Thread(target=mfrc_rfid_listener, args=(reader2, "Signal2")).start()
threading.Thread(target=traffic_light_controller).start()
threading.Thread(target=stop_listener).start()

# Main loop to keep program alive
try:
    while not shutdown:
        time.sleep(1)
finally:
    GPIO.cleanup()
    if serial_rfid1:
        serial_rfid1.close()
    print("Program stopped and GPIO cleaned up.")
